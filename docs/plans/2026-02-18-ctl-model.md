# /ctl/model Endpoint Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a `GET /ctl/model` endpoint that discovers a local LM Studio server, queries downloaded and loaded models, merges the two lists, and renders a static HTML table for logged-in users.

**Architecture:** Single new file `src/cmd/web/lmstudio.go` with a pure `mergeModels` helper and the `modelCtl` handler. A package-level `lmstudioDiscover` var makes discovery injectable for tests. A new embedded template `src/internal/assets/html/ctl-model.html` renders the table. Route registered under `requireAuth` in `main.go`.

**Tech Stack:** `github.com/hypernetix/lmstudio-go/pkg/lmstudio` (client + mock), `github.com/dustin/go-humanize` (size formatting), Go's `net/http/httptest` (handler tests), embedded HTML templates.

```go
// mergeModels combines downloaded and loaded model lists, deduplicating by
// ModelKey. Downloaded models that also appear in the loaded list have their
// IsLoaded field set. Loaded models absent from the downloaded list are
// appended. Input slices and their elements are not mutated.
func mergeModels(downloaded, loaded []*lmstudio.Model) []*lmstudio.Model {
	loadedKeys := make(map[string]bool, len(loaded))
	for _, m := range loaded {
		loadedKeys[m.ModelKey] = true
	}

	result := make([]*lmstudio.Model, 0, len(downloaded)+len(loaded))
	downloadedKeys := make(map[string]bool, len(downloaded))

	for _, m := range downloaded {
		downloadedKeys[m.ModelKey] = true
		c := *m
		c.IsLoaded = loadedKeys[m.ModelKey]
		result = append(result, &c)
	}
	for _, m := range loaded {
		if !downloadedKeys[m.ModelKey] {
			c := *m
			c.IsLoaded = true
			result = append(result, &c)
		}
	}
	return result
}
```

```go
package main

import (
	"html/template"
	"net/http"
	"strings"

	"github.com/dustin/go-humanize"
	"github.com/hypernetix/lmstudio-go/pkg/lmstudio"

	"github.com/refinement-systems/BorealValley/src/internal/assets"
)

// lmstudioDiscover is a package-level variable so tests can replace it without
// starting a real LM Studio server.
var lmstudioDiscover = func() (string, error) {
	log := lmstudio.NewLogger(lmstudio.LogLevelError)
	return lmstudio.DiscoverLMStudioServer("", 0, log)
}

type modelCtlData struct {
	Models []modelRow
	Err    string
}

type modelRow struct {
	Name     string
	Type     string
	Format   string
	Size     string
	IsLoaded bool
}

var modelCtlTmpl = template.Must(template.New("ctl-model").Parse(assets.HtmlCtlModel))

// mergeModels combines downloaded and loaded model lists, deduplicating by
// ModelKey. Downloaded models that also appear in the loaded list have their
// IsLoaded field set. Loaded models absent from the downloaded list are
// appended. Input slices and their elements are not mutated.
func mergeModels(downloaded, loaded []*lmstudio.Model) []*lmstudio.Model {
	loadedKeys := make(map[string]bool, len(loaded))
	for _, m := range loaded {
		loadedKeys[m.ModelKey] = true
	}

	result := make([]*lmstudio.Model, 0, len(downloaded)+len(loaded))
	downloadedKeys := make(map[string]bool, len(downloaded))

	for _, m := range downloaded {
		downloadedKeys[m.ModelKey] = true
		c := *m
		c.IsLoaded = loadedKeys[m.ModelKey]
		result = append(result, &c)
	}
	for _, m := range loaded {
		if !downloadedKeys[m.ModelKey] {
			c := *m
			c.IsLoaded = true
			result = append(result, &c)
		}
	}
	return result
}

func toModelRows(models []*lmstudio.Model) []modelRow {
	rows := make([]modelRow, 0, len(models))
	for _, m := range models {
		name := m.ModelName
		if name == "" {
			name = m.ModelKey
		}
		var size string
		if m.Size > 0 {
			size = humanize.Bytes(uint64(m.Size))
		}
		rows = append(rows, modelRow{
			Name:     name,
			Type:     m.Type,
			Format:   m.Format,
			Size:     size,
			IsLoaded: m.IsLoaded,
		})
	}
	return rows
}

// modelCtl handles GET /ctl/model: discovers LM Studio, queries both the
// downloaded and loaded model lists concurrently, merges them, and renders the
// result. Partial results are shown alongside any error message.
func (app *application) modelCtl(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodGet {
		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}

	data := modelCtlData{}

	addr, err := lmstudioDiscover()
	if err != nil {
		data.Err = "LM Studio not found: " + err.Error()
		renderModelCtl(w, data)
		return
	}

	log := lmstudio.NewLogger(lmstudio.LogLevelError)
	client := lmstudio.NewLMStudioClient(addr, log)
	defer client.Close()

	type queryResult struct {
		models []*lmstudio.Model
		err    error
	}
	downloadedCh := make(chan queryResult, 1)
	loadedCh := make(chan queryResult, 1)

	go func() {
		m, err := client.ListDownloadedModels()
		downloadedCh <- queryResult{m, err}
	}()
	go func() {
		m, err := client.ListAllLoadedModels()
		loadedCh <- queryResult{m, err}
	}()

	dRes := <-downloadedCh
	lRes := <-loadedCh

	var errs []string
	if dRes.err != nil {
		errs = append(errs, "downloaded: "+dRes.err.Error())
	}
	if lRes.err != nil {
		errs = append(errs, "loaded: "+lRes.err.Error())
	}
	if len(errs) > 0 {
		data.Err = strings.Join(errs, "; ")
	}

	data.Models = toModelRows(mergeModels(dRes.models, lRes.models))
	renderModelCtl(w, data)
}

func renderModelCtl(w http.ResponseWriter, data modelCtlData) {
	w.Header().Set("Content-Type", "text/html; charset=utf-8")
	_ = modelCtlTmpl.Execute(w, data)
}
```
