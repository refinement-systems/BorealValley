# /ctl/chat Agent Chat Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a `/ctl/chat` endpoint that lets a logged-in user submit a prompt to a loaded LM Studio model, with the server running a full agent loop (tool calls for `list_dir`, `read_file`, `search_text`) before returning the final answer.

**Architecture:** Two new source files (`chat.go` for the HTTP handler + agent loop, `tools.go` for sandboxed file tools), one new HTML template, and small additions to `assets.go` and `main.go`. The agent loop uses Go's `net/http` to call LM Studio's OpenAI-compatible `/v1/chat/completions` endpoint; the existing `lmstudio-go` package is used for server discovery and listing loaded models.

**Tech Stack:** Go stdlib `net/http`, `encoding/json`, `path/filepath`, `regexp`; `github.com/hypernetix/lmstudio-go` (discovery + model list); Go `html/template`.

package main

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"regexp"
	"strings"
)

// sandboxed returns the absolute path of userPath joined to root.
// It returns an error if the result would escape root.
func sandboxed(root, userPath string) (string, error) {
	abs := filepath.Clean(filepath.Join(root, userPath))
	if abs != root && !strings.HasPrefix(abs, root+string(filepath.Separator)) {
		return "", fmt.Errorf("path %q escapes sandbox", userPath)
	}
	return abs, nil
}

type dirEntry struct {
	Name  string `json:"name"`
	IsDir bool   `json:"isDir"`
	Size  int64  `json:"size,omitempty"`
}

// listDir returns a JSON array of entries at path (relative to root).
func listDir(root, path string) (string, error) {
	abs, err := sandboxed(root, path)
	if err != nil {
		return "", err
	}
	entries, err := os.ReadDir(abs)
	if err != nil {
		return "", err
	}
	rows := make([]dirEntry, 0, len(entries))
	for _, e := range entries {
		var size int64
		if !e.IsDir() {
			if info, err := e.Info(); err == nil {
				size = info.Size()
			}
		}
		rows = append(rows, dirEntry{Name: e.Name(), IsDir: e.IsDir(), Size: size})
	}
	b, _ := json.Marshal(rows)
	return string(b), nil
}

// readFile returns the content of path (relative to root), capped at 32 KB.
func readFile(root, path string) (string, error) {
	abs, err := sandboxed(root, path)
	if err != nil {
		return "", err
	}
	data, err := os.ReadFile(abs)
	if err != nil {
		return "", err
	}
	const maxBytes = 32 * 1024
	if len(data) > maxBytes {
		data = data[:maxBytes]
	}
	return string(data), nil
}

type searchMatch struct {
	File string `json:"file"`
	Line int    `json:"line"`
	Text string `json:"text"`
}

// searchText walks the tree at path (relative to root), regex-matches each line
// against query, and returns JSON of up to 100 matches.
func searchText(root, path, query string) (string, error) {
	abs, err := sandboxed(root, path)
	if err != nil {
		return "", err
	}
	re, err := regexp.Compile(query)
	if err != nil {
		return "", fmt.Errorf("invalid query: %w", err)
	}
	var matches []searchMatch
	const maxMatches = 100
	err = filepath.WalkDir(abs, func(p string, d os.DirEntry, err error) error {
		if err != nil || d.IsDir() {
			return err
		}
		data, err := os.ReadFile(p)
		if err != nil {
			return nil // skip unreadable files
		}
		rel, _ := filepath.Rel(root, p)
		for i, line := range strings.Split(string(data), "\n") {
			if re.MatchString(line) {
				matches = append(matches, searchMatch{
					File: rel,
					Line: i + 1,
					Text: strings.TrimSpace(line),
				})
				if len(matches) >= maxMatches {
					return filepath.SkipAll
				}
			}
		}
		return nil
	})
	if err != nil {
		return "", err
	}
	if matches == nil {
		return "[]", nil
	}
	b, _ := json.Marshal(matches)
	return string(b), nil
}

package main

import (
	"bytes"
	"encoding/json"

	"github.com/hypernetix/lmstudio-go/pkg/lmstudio"

)

// OpenAI-compatible request/response types.

type chatMessage struct {
	Role       string     `json:"role"`
	Content    string     `json:"content,omitempty"`
	ToolCalls  []toolCall `json:"tool_calls,omitempty"`
	ToolCallID string     `json:"tool_call_id,omitempty"`
}

type toolCall struct {
	ID       string       `json:"id"`
	Type     string       `json:"type"`
	Function toolCallFunc `json:"function"`
}

type toolCallFunc struct {
	Name      string `json:"name"`
	Arguments string `json:"arguments"`
}

type chatRequest struct {
	Model       string        `json:"model"`
	Messages    []chatMessage `json:"messages"`
	Tools       []toolDef     `json:"tools,omitempty"`
	Temperature float64       `json:"temperature"`
}

type toolDef struct {
	Type     string      `json:"type"`
	Function toolFuncDef `json:"function"`
}

type toolFuncDef struct {
	Name        string          `json:"name"`
	Description string          `json:"description"`
	Parameters  json.RawMessage `json:"parameters"`
}

type chatChoice struct {
	Message      chatMessage `json:"message"`
	FinishReason string      `json:"finish_reason"`
}

type chatResponse struct {
	Choices []chatChoice `json:"choices"`
}

// Template data.

type toolCallTrace struct {
	Name   string
	Args   string
	Result string
}

type chatCtlData struct {
	Models    []modelRow
	Prompt    string
	Result    string
	ToolCalls []toolCallTrace
	Err       string
}

var chatTmpl = template.Must(template.New("ctl-chat").Parse(assets.HtmlCtlChat))

// chatCtl handles GET and POST /ctl/chat.
func (app *application) chatCtl(w http.ResponseWriter, r *http.Request) {
	switch r.Method {
	case http.MethodGet:
		app.chatCtlGet(w, r)
	case http.MethodPost:
		app.chatCtlPost(w, r)
	default:
		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
	}
}

func (app *application) chatCtlGet(w http.ResponseWriter, r *http.Request) {
	data := chatCtlData{}

	addr, err := lmstudioDiscover()
	if err != nil {
		data.Err = "LM Studio not found: " + err.Error()
		renderChatCtl(w, data)
		return
	}

	log := lmstudio.NewLogger(lmstudio.LogLevelError)
	client := lmstudio.NewLMStudioClient(addr, log)
	defer client.Close()

	loaded, err := client.ListAllLoadedModels()
	if err != nil {
		data.Err = "failed to list models: " + err.Error()
		renderChatCtl(w, data)
		return
	}

	for _, m := range loaded {
		if m.Type != "" && m.Type != "llm" {
			continue
		}
		id := m.Identifier
		if id == "" {
			id = m.ModelKey
		}
		name := m.DisplayName
		if name == "" {
			name = m.ModelName
		}
		if name == "" {
			name = id
		}
		data.Models = append(data.Models, modelRow{Name: name, Key: id})
	}

	renderChatCtl(w, data)
}

func (app *application) chatCtlPost(w http.ResponseWriter, r *http.Request) {
	if err := r.ParseForm(); err != nil {
		http.Error(w, "bad form", http.StatusBadRequest)
		return
	}
	model := r.PostFormValue("model")
	prompt := r.PostFormValue("prompt")

	data := chatCtlData{Prompt: prompt}

	if model == "" || prompt == "" {
		data.Err = "model and prompt are required"
		renderChatCtl(w, data)
		return
	}

	addr, err := lmstudioDiscover()
	if err != nil {
		data.Err = "LM Studio not found: " + err.Error()
		renderChatCtl(w, data)
		return
	}

	result, traces, err := app.runAgentLoop(addr, model, prompt)
	data.Result = result
	data.ToolCalls = traces
	if err != nil {
		data.Err = err.Error()
	}

	renderChatCtl(w, data)
}

// runAgentLoop sends messages to LM Studio's OpenAI-compatible chat completions
// endpoint, executing tool calls until the model stops or the iteration limit is hit.
func (app *application) runAgentLoop(baseURL, model, prompt string) (string, []toolCallTrace, error) {
	const maxIter = 10

	tools := agentTools()
	messages := []chatMessage{
		{Role: "system", Content: "You are a helpful coding assistant. Use the provided tools to inspect the workspace when needed."},
		{Role: "user", Content: prompt},
	}

	httpClient := &http.Client{}
	var traces []toolCallTrace

	for i := 0; i < maxIter; i++ {
		reqBody, err := json.Marshal(chatRequest{
			Model:       model,
			Messages:    messages,
			Tools:       tools,
			Temperature: 0.2,
		})
		if err != nil {
			return "", traces, fmt.Errorf("failed to marshal request: %w", err)
		}

		resp, err := httpClient.Post(baseURL+"/v1/chat/completions", "application/json", bytes.NewReader(reqBody))
		if err != nil {
			return "", traces, fmt.Errorf("chat request failed: %w", err)
		}
		body, err := io.ReadAll(resp.Body)
		resp.Body.Close()
		if err != nil {
			return "", traces, fmt.Errorf("failed to read response: %w", err)
		}

		var cr chatResponse
		if err := json.Unmarshal(body, &cr); err != nil {
			return "", traces, fmt.Errorf("failed to decode response: %w", err)
		}
		if len(cr.Choices) == 0 {
			return "", traces, fmt.Errorf("no choices in response")
		}

		choice := cr.Choices[0]

		if choice.FinishReason == "stop" || choice.FinishReason == "" {
			return choice.Message.Content, traces, nil
		}

		if choice.FinishReason == "tool_calls" {
			messages = append(messages, choice.Message)
			for _, tc := range choice.Message.ToolCalls {
				toolResult := app.executeTool(tc)
				traces = append(traces, toolCallTrace{
					Name:   tc.Function.Name,
					Args:   tc.Function.Arguments,
					Result: toolResult,
				})
				messages = append(messages, chatMessage{
					Role:       "tool",
					Content:    toolResult,
					ToolCallID: tc.ID,
				})
			}
			continue
		}

		// Unexpected finish_reason — return what we have.
		return choice.Message.Content, traces, nil
	}

	return "", traces, fmt.Errorf("agent loop hit iteration limit (%d)", maxIter)
}

// executeTool dispatches a single tool call and returns the result as a string.
func (app *application) executeTool(tc toolCall) string {
	var args map[string]string
	if err := json.Unmarshal([]byte(tc.Function.Arguments), &args); err != nil {
		return `{"error": "failed to parse arguments"}`
	}

	var result string
	var err error

	switch tc.Function.Name {
	case "list_dir":
		result, err = listDir(app.repoRoot, args["path"])
	case "read_file":
		result, err = readFile(app.repoRoot, args["path"])
	case "search_text":
		result, err = searchText(app.repoRoot, args["path"], args["query"])
	default:
		return fmt.Sprintf(`{"error": "unknown tool %q"}`, tc.Function.Name)
	}

	if err != nil {
		return fmt.Sprintf(`{"error": %q}`, err.Error())
	}
	return result
}
